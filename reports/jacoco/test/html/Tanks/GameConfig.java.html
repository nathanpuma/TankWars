package Tanks;

import processing.core.PApplet;
import processing.core.PImage;
import processing.data.JSONArray;
import processing.data.JSONObject;

public class GameConfig {
    private PApplet app;
    private JSONObject config;
    private PImage background;
    private PImage treeImage; // This should be a member variable now
    private JSONObject playerColors; // To store player color settings
    private String layoutFilename;
    private int foregroundColour;
    private int currentLevelIndex = 0;
    private JSONArray levels;

    public GameConfig(PApplet app, String configPath) {
        this.app = app;
        this.config = app.loadJSONObject(configPath);
        this.levels = config.getJSONArray("levels");
        this.playerColors = config.getJSONObject("player_colours");
        loadLevel(currentLevelIndex);
    }
    

    public void loadNextLevel() {
        currentLevelIndex = (currentLevelIndex + 1) % levels.size(); // Loop back to the first level if it's the last
        loadLevel(currentLevelIndex);
    }

    public boolean hasNextLevel() {
        return currentLevelIndex + 1 < levels.size();
    }

    public void resetGame() {
        currentLevelIndex = -1; // Explicitly set to the first level
        loadLevel(currentLevelIndex); // Reload the first level directly
    }
    
    
    
    private void loadLevel(int levelIndex) {
        if (levelIndex < 0 || levelIndex >= levels.size()) {
            levelIndex = 0;  // Reset to first level if index is out of range
        }
        currentLevelIndex = levelIndex;

        JSONArray levels = config.getJSONArray("levels");
        JSONObject level = levels.getJSONObject(levelIndex);

        String backgroundFilename = level.getString("background");
        this.background = app.loadImage(backgroundFilename);

        this.layoutFilename = level.getString("layout");

        // Check if the level has trees defined, if not, set treeImage to null
        if (level.hasKey("trees")) {
            this.treeImage = app.loadImage(level.getString("trees")); // Load the tree image
        } else {
            this.treeImage = null; // No tree image for this level
        }

        String foregroundColourString = level.getString("foreground-colour");
        this.foregroundColour = parseColourString(foregroundColourString);
    }

    private int parseColourString(String colorString) {
        String[] colorValues = colorString.split(",");
        return app.color(Integer.parseInt(colorValues[0].trim()),
                         Integer.parseInt(colorValues[1].trim()),
                         Integer.parseInt(colorValues[2].trim()));
    }

    public PImage getBackground() {
        return background;
    }

    public String getLayoutFilename() {
        return layoutFilename;
    }

    public PImage getTreeImage() {
        return treeImage;
    }

    public int getForegroundColour() {
        return foregroundColour;
    }

    public int getPlayerColor(char playerID) {
        String key = String.valueOf(playerID);
        if (playerColors.hasKey(key)) {
            String colorStr = playerColors.getString(key);
            if (colorStr.equals("random")) {
                return app.color((int)app.random(256), (int)app.random(256), (int)app.random(256));
            } else {
                return parseColourString(colorStr);
            }
        }
        return app.color(200); // Default color if no specific color is defined
    }

    public void displayBackground(int width, int height) {
        app.image(background, 0, 0, width, height);
    }
}
